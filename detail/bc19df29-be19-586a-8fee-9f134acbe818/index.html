void my_replace(string& s, const string& old_val, const string& new_val) {
    string::iterator iter1, iter2;
    iter1 = s.begin();
    while (iter1 != s.end()) {
        while (*iter1 == ' ')
            iter1++;
        iter2 = iter1;
        while (iter2 != s.end() && *iter2 != ' ')
            iter2++;
        string sub(iter1, iter2);
        if (sub == old_val) {
            iter2 = s.erase(iter1, iter2);
            s.insert(iter2 - s.begin(), new_val);
        }
        iter1 = iter2;
    }
}
